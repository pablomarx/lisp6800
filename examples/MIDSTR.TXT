
(DEF MIDSTR (LAMBDA (STR NUM)
	(MIDSTR1 STR NUM (LENGTH STR))
))

(DEF MIDSTR1 (LAMBDA (STR NUM LEN)
	(COND
	((GREATER 1 LEN) NIL)
	((GREATER NUM LEN) NIL)
	(T (MIDSTR2 STR NUM))
	)
))

(DEF MIDSTR2 (LAMBDA (STR NUM)
	(COND
	((EQ NUM 1) (CAR STR))
	(T (MIDSTR2 (CDR STR) (SUB NUM 1)))
	)
))
